\item Open source implementation
\end{itemize}

\begin{frame}
\frametitle{Architecture}
\begin{block}{Conversion Pipeline}
\begin{enumerate}
\item Parse graphical model
\item Convert to MRF (moralization if needed)
\item Find maximal cliques
\item Encode as Ising Hamiltonian
\item Generate framework-specific code
\end{enumerate}
\end{block}
\end{frame}
